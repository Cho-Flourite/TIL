### 별 찍기




```
global n
n = int(input())
global arr
arr = [['*' for _ in range(n)] for _ in range(n)]

def rstar(N):
    global arr, n
    if N == n:
        return;
    for k in range(0,n-1,N*3):
        for i in range(N+k,N*2+k):
            for l in range(0,n-1,N*3):
                for j in range(N+l,N*2+l):
                    arr[i][j] = ' '
    rstar(N*3)

def star(N):
    global arr ,n
    rstar(N)
    for i in arr:
        for j in i:
            print(j, end='')
        print()
        
star(1)
```
