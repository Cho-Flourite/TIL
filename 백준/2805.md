n,M = map(int,input().split())
arr = list(map(int,input().split()))

s = 0
e = max(arr)
m = (s+e)//2
while True:
    sum=0
    if e - s == 1:
        print(m)
        break
    for i in arr:
        if i > m:
            sum += i - m
    if sum < M:
        e = m
        m = (s+m)//2
    elif sum > M:
        s = m
        m = (e+m)//2
    else:
        print(m)
        break
