import sys

input = sys.stdin.readline
n, m, b = map(int, input().split())
arr = []

for i in range(n):
    arr.append(list(map(int, input().split())))
mmin = min(min(arr))
mmax = max(max(arr))

def findM(ar, m, b):
    block = b
    sum = 0
    for j in ar:
        for i in j:
            if i >= m:
                sum += (i - m)*2
                block += i-m
            else:
                sum += m - i
                block -= m-i
    if block < 0:
        return -1
    return sum

minSum = findM(arr, mmin, b)
maxGround = mmin
for i in range(mmin,mmax+1):
    sum = findM(arr, i, b)

    if minSum > sum and sum != -1:
        
        minSum = sum
        maxGround = i
        
    elif minSum == sum and sum != -1:
        maxGround = i

print(minSum,maxGround)
